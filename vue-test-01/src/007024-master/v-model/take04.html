	computed:{
		compute01(){
			console.log("com01");
			this.uniqueSet = new Set(this.array);
			this.backToArray = [...this.uniqueSet]
		},
		compute02(){
			console.log("com02");
			this.arrayfrom = Array.from(new Set(this.array));
		},
		compute03(){
			console.log("com03");
			this.arrayfilter = this.array.filter((item, index) => {
				console.log(
					//a. item
					item,
					//b. index
					index,
					//c. indexOf
					this.array.indexOf(item),
					//d, condition
					this.array.indexOf(item) === index
				);
				return this.array.indexOf(item) === index;
			});
		},
		compute04(){
			console.log("com04");
			this.arrayfilterdu = this.array.filter((item, index) => {
			return this.array.indexOf(item) !== index
			});
		},
		compute05(){
			console.log("com05");
			this.arrayreduce = this.array.reduce((unique, item) => {
				console.log(
					//a. Item
					item,
					//b. Final Array (Accumulator)
					unique,
					//c. Condition (Remember it only get pushed if this returns false)
					unique.includes(item),
					//d, Reducer Function Result
					unique.includes(item) ? unique : [...unique, item]
				);
				return unique.includes(item) ? unique : [...unique, item];
			});
		}
	}
